""" return a + b > c and a + c > b and b + c > a # Ingreso de datos por teclado lado1 = float(input("Ingrese el primer lado del triángulo: ")) lado2 = float(input("Ingrese el segundo lado del triángulo: ")) lado3 = float(input("Ingrese el tercer lado del triángulo: ")) # Llamada a la función e impresión del resultado resultado = esTriangulo(lado1, lado2, lado3) print(f"¿Es un triángulo?: {resultado}")
